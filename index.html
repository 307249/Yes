const firebaseConfig = { databaseURL: "https://drosak-v2-default-rtdb.europe-west1.firebasedatabase.app" }; function showPage(id) { document.querySelectorAll('.page').forEach(p => p.classList.remove('active')); document.getElementById(id).classList.add('active'); } async function handleAccess() { const input = document.getElementById("codeInput"); const code = input.value.trim(); const errorBox = document.getElementById("errorMsg"); const now = Date.now(); try { // جلب حالة التفعيل const lockRes = await fetch(`${firebaseConfig.databaseURL}/appSettings/lockEnabled.json`); const isLocked = await lockRes.json(); console.log("lockEnabled = ", isLocked); // لمراقبة القيمة في console if (isLocked !== true) { showPage("subjectsPage"); return; } // إذا القفل مفعل، تحقق من الكود const keysRes = await fetch(`${firebaseConfig.databaseURL}/validKeys.json`); const keys = await keysRes.json(); const savedKey = localStorage.getItem("drosakKey"); let validKey = null; for (const key in keys) { const k = keys[key]; if ((key === code || key === savedKey) && now < k.expiresAt) { validKey = key; break; } } if (validKey) { localStorage.setItem("drosakKey", validKey); showPage("subjectsPage"); } else { if (code && keys[code] && now >= keys[code].expiresAt) { errorBox.textContent = "⚠️ انتهت صلاحية الكود الخاص بك. للتجديد كلمنا: @AL_MAALA"; } else { errorBox.textContent = "❌ الكود غير صحيح."; } } } catch (err) { console.error("Error:", err); errorBox.textContent = "⚠️ حدث خطأ في الاتصال بقاعدة البيانات."; } }
